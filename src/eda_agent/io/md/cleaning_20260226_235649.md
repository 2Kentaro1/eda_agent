```python
import pandas as pd
import numpy as np

# DataFrameの再構築 (プレビューデータに基づいて)
data = {
    'datetime': ['2013-11-18', '2013-11-19', '2013-11-20', '2013-11-21', '2013-11-22'],
    'y': [90, 101, 118, 120, 130],
    'week': ['月', '火', '水', '木', '金'],
    'soldout': [0, 1, 0, 1, 1],
    'name': ['厚切りイカフライ', '手作りヒレカツ', '白身魚唐揚げ野菜あん', '若鶏ピリ辛焼', 'ビッグメンチカツ'],
    'kcal': [np.nan, np.nan, np.nan, np.nan, np.nan],
    'remarks': [np.nan, np.nan, np.nan, np.nan, np.nan],
    'event': [np.nan, np.nan, np.nan, np.nan, np.nan],
    'payday': [np.nan, np.nan, np.nan, np.nan, np.nan],
    'weather': ['快晴', '快晴', '快晴', '快晴', '快晴'],
    'precipitation': ['--', '--', '--', '--', '--'],
    'temperature': [19.8, 17.0, 15.5, 15.2, 16.1]
}
df = pd.DataFrame(data)

# データクリーニングの実行

# 1. 'datetime' 列: datetime型に変換
df['datetime'] = pd.to_datetime(df['datetime'])

# 2. 'y' 列: 目的変数。数値型であることを確認。
#    すでに数値型（int）のように見えるため、ここでは特に変換は不要。

# 3. 'week' 列: 曜日。カテゴリ型として扱う。
#    データ型はobjectで問題ない。

# 4. 'soldout' 列: 売り切れフラグ。数値型（0または1）であることを確認。
df['soldout'] = df['soldout'].astype(int)

# 5. 'name' 列: 商品名。文字列の前後空白除去や小文字化は、必要であれば実行するが、
#    現状のプレビューでは不要と判断。
df['name'] = df['name'].astype(str) # 念のため文字列型に

# 6. 'kcal' 列: カロリー。数値型に変換し、欠損値はNaNのまま保持。
#    'nan'という文字列があればNaNに変換するが、np.nanなのでそのまま。
df['kcal'] = pd.to_numeric(df['kcal'], errors='coerce')

# 7. 'remarks' 列: 備考。欠損値はNaNのまま保持。
#    カテゴリカルデータとして扱う。
#    `nan`はPandasによってNone/np.nanとして認識されるため、特別な変換は不要。

# 8. 'event' 列: イベント。欠損値はNaNのまま保持。
#    カテゴリカルデータとして扱う。
#    `nan`はPandasによってNone/np.nanとして認識されるため、特別な変換は不要。

# 9. 'payday' 列: 給料日フラグ。NaNを0に置き換え、int型に変換。
#    プレビューでは全てNaNだが、実際のデータでは1も含まれると想定。
df['payday'] = df['payday'].fillna(0).astype(int)

# 10. 'weather' 列: 天気。カテゴリカルデータとして扱う。
#     データ型はobjectで問題ない。

# 11. 'precipitation' 列: 降水量。'--'を0に置き換え、数値型に変換。
#     数値に変換できない値をNaNにするためにerrors='coerce'を使用。
df['precipitation'] = df['precipitation'].replace('--', '0').astype(float)

# 12. 'temperature' 列: 気温。数値型であることを確認。
df['temperature'] = pd.to_numeric(df['temperature'], errors='coerce')


# クリーニング後のDataFrameの確認
print("--- クリーニング後のDataFrame先頭5行 ---")
print(df.head())
print("\n--- クリーニング後のDataFrame情報 ---")
print(df.info())
print("\n--- クリーニング後の各列のユニーク値（一部） ---")
for col in df.columns:
    if df[col].dtype == 'object' or df[col].nunique() < 10: # 数値型でユニーク値が少ないものも含む
        print(f"'{col}': {df[col].unique()}")
```
